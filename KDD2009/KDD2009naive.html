# naively transform data
cleanColumn <- function(c,targetSet) {
  if(is.logical(c)) {
    c <- as.numeric(c)
  }
  if(is.numeric(c)) {
    c[is.na(c)] <- 0
  }
  if(is.factor(c)) {
    c <- as.character(c)
  }
  if(is.character(c)) {
    c[is.na(c)] <- ''
    if(!is.null(targetSet)) {
      c[!(c %in% targetSet)] <- ''
    }
  }
  c
}
